<Steps>
  <Step num="1" title="Zabezpiecz wewnętrzny obwód">

  <Step num="2" title="Ustanów południowe przyczółki">

  </Step>
  <Step num="3" title="Połącz wschodnie wybrzeże">
    Poleć do Juliet 1 i użyj przybrzeżnych linii drzew, aby bezpiecznie dojść do Ban Pa (India 1/2) i Blue Lagoon (Kilo 1/2). Unikaj głównych dróg, aby ominąć punkty kontrolne AI.
  </Step>
  <Step num="4" title="Zinfiltruj północne twierdze">
