<Steps>
  <Step num="1" title="Sichern Sie den inneren Perimeter">
    Beginnen Sie damit, Echo 1 und Echo 2 rund um das Sägewerk freizuschalten. Diese Zonen bieten eine sichere Pufferzone zwischen Ihrer Fraktionsbasis und dem stark umkämpften Zentrum der Karte.
  </Step>

  </Step>
  <Step num="3" title="Verbinden Sie die Ostküste">

  <Step num="4" title="Infiltrieren Sie nördliche Festungen">

  </Step>
</Steps>
